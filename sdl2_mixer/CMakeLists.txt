cmake_minimum_required(VERSION 3.0)

set(CMAKE_TOOLCHAIN_FILE "$ENV{DOLCESDK}/share/dolce.toolchain.cmake" CACHE PATH "DolceSDK toolchain file")

project(sdl2_mixer LANGUAGES C)

add_compile_options(
  -O3 -Wall
)

add_compile_definitions(
  MUSIC_FLAC
  MUSIC_MOD_MIKMOD
  MUSIC_MP3_MPG123
  MUSIC_OGG
  MUSIC_WAV
)

include_directories("${CMAKE_INSTALL_PREFIX}/include/SDL2")

add_library(SDL2_mixer STATIC
  effect_position.c
  effect_stereoreverse.c
  effects_internal.c
  load_aiff.c
  load_voc.c
  mixer.c
  music.c
  music_cmd.c
  music_flac.c
  music_fluidsynth.c
  music_mad.c
  music_mikmod.c
  music_modplug.c
  music_mpg123.c
  music_nativemidi.c
  music_ogg.c
  music_opus.c
  music_timidity.c
  music_wav.c
)

install(TARGETS SDL2_mixer DESTINATION lib)
install(FILES SDL_mixer.h DESTINATION include/SDL2)
